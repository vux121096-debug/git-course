2. Hãy mô tả cấu trúc html của 1 bảng
    <table> // Create table
        <thead> // Table Head
            <tr> // Row 0
                <th>Table Header A</th>
                <th>Table Header B</th>
                <th>Table Header C</th>
            </tr>
        </thead>
        <tbody> // Table body
            <tr> // Row 1
                <td>Table data A1</td>
                <td>Table data B1</td>
                <td>Table data C1</td>
            </tr>
            <tr> // Row 2
                <td>Table data A2</td>
                <td>Table data B2</td>
                <td>Table data C2</td>
            </tr>
            <tr> // Row 3
                <td>Table data A3</td>
                <td>Table data B3</td>
                <td>Table data C3</td>
            </tr>
        </tbody>
        <tfoot> // Table Footer
            <tr> 
                <td colspan="3">Summary</td>
            </tr>
        </tfoot>
    </table>
